FOR THIS WE NEED TO ADD VENV(VIRTUAL ENVIRONMENT)

# Navigate to your project directory
cd path\to\your\project

# Create a virtual environment named "venv"
python -m venv venv

# Activate the virtual environment
venv\Scripts\activate
